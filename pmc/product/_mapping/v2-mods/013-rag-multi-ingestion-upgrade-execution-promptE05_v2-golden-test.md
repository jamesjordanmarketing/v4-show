# Vercel-Based Golden-Set Regression Test Specification

## Problem Statement

The CLI-based golden-set test (`scripts/golden-set-test.js`) requires:
1. Manual cookie extraction from browser DevTools
2. Dev server to be running locally
3. Complex environment variable setup
4. All 20 queries returned `500: Internal Server Error` when tested

**Solution:** Build a Vercel API route that runs the golden-set test server-side with proper authentication and error handling.

---

## Architecture

### New Files to Create

1. **`src/app/api/rag/test/golden-set/route.ts`** — Main test runner API route
2. **`src/lib/rag/testing/golden-set-definitions.ts`** — Shared golden-set Q&A pairs
3. **`src/app/rag/test/page.tsx`** — UI page to trigger and view test results

## File 2: Test Runner API Route

**Path:** `src/app/api/rag/test/golden-set/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { requireAuth } from '@/lib/supabase-server';
import { queryRAG } from '@/lib/rag/services/rag-retrieval-service';
import { GOLDEN_SET, CANONICAL_DOCUMENT_ID, TARGET_PASS_RATE } from '@/lib/rag/testing/golden-set-definitions';

export const maxDuration = 300; // 5 minutes for all 20 queries

interface TestResult {
  question: string;
  difficulty: string;
  passed: boolean;
  expectedAnswer: string;
  actualResponse: string;
  selfEvalScore: number | null;
  responseTimeMs: number;
  error?: string;
}

export async function POST(request: NextRequest) {
  try {
    const { user, response } = await requireAuth(request);
    if (response) return response;

    const results: TestResult[] = [];
    let passed = 0;
    let failed = 0;

    console.log('[Golden-Set Test] Starting test run for user:', user.id);

    for (let i = 0; i < GOLDEN_SET.length; i++) {
      const item = GOLDEN_SET[i];
      const startTime = Date.now();

      try {
        // Call the RAG service directly (server-side, no HTTP overhead)
        const result = await queryRAG({
          queryText: item.question,
          documentId: CANONICAL_DOCUMENT_ID,
          mode: 'rag_only',
          userId: user.id,
        });

        const responseTimeMs = Date.now() - startTime;
        const responseText = result.query.responseText || '';
        const selfEvalScore = result.query.selfEvalScore;
        const containsExpected = responseText.toLowerCase().includes(item.expectedAnswer.toLowerCase());

        results.push({
          question: item.question,
          difficulty: item.difficulty,
          passed: containsExpected,
          expectedAnswer: item.expectedAnswer,
          actualResponse: responseText.slice(0, 200),
          selfEvalScore,
          responseTimeMs,
        });

        if (containsExpected) {
          passed++;
          console.log(`  ✓ [${item.difficulty}] ${item.question}`);
        } else {
          failed++;
          console.log(`  ✗ [${item.difficulty}] ${item.question}`);
          console.log(`    Expected: "${item.expectedAnswer}"`);
          console.log(`    Got: "${responseText.slice(0, 150)}..."`);
        }
      } catch (err) {
        failed++;
        const errorMsg = err instanceof Error ? err.message : 'Unknown error';
        console.error(`  ✗ [${item.difficulty}] ${item.question} - ERROR:`, errorMsg);

        results.push({
          question: item.question,
          difficulty: item.difficulty,
          passed: false,
          expectedAnswer: item.expectedAnswer,
          actualResponse: '',
          selfEvalScore: null,
          responseTimeMs: Date.now() - startTime,
          error: errorMsg,
        });
      }

      // Small delay between queries to avoid overwhelming the system
      await new Promise(resolve => setTimeout(resolve, 500));
    }

    // Calculate summary statistics
    const total = passed + failed;
    const passRate = ((passed / total) * 100).toFixed(1);

    const easyPassed = results.filter(r => r.difficulty === 'easy' && r.passed).length;
    const easyTotal = results.filter(r => r.difficulty === 'easy').length;
    const mediumPassed = results.filter(r => r.difficulty === 'medium' && r.passed).length;
    const mediumTotal = results.filter(r => r.difficulty === 'medium').length;
    const hardPassed = results.filter(r => r.difficulty === 'hard' && r.passed).length;
    const hardTotal = results.filter(r => r.difficulty === 'hard').length;

    const avgResponseTime = results.reduce((sum, r) => sum + r.responseTimeMs, 0) / results.length;
    const scoresWithValues = results.filter(r => r.selfEvalScore !== null);
    const avgSelfEval = scoresWithValues.length > 0
      ? scoresWithValues.reduce((sum, r) => sum + (r.selfEvalScore || 0), 0) / scoresWithValues.length
      : 0;

    const summary = {
      totalPassed: passed,
      totalFailed: failed,
      passRate: parseFloat(passRate),
      meetsTarget: parseFloat(passRate) >= TARGET_PASS_RATE,
      breakdown: {
        easy: { passed: easyPassed, total: easyTotal },
        medium: { passed: mediumPassed, total: mediumTotal },
        hard: { passed: hardPassed, total: hardTotal },
      },
      avgResponseTimeMs: Math.round(avgResponseTime),
      avgSelfEvalScore: Math.round(avgSelfEval * 100) / 100,
    };

    console.log('[Golden-Set Test] Complete:', summary);

    return NextResponse.json({
      success: true,
      data: {
        summary,
        results,
      },
    });
  } catch (err) {
    console.error('[Golden-Set Test] Fatal error:', err);
    return NextResponse.json(
      { success: false, error: err instanceof Error ? err.message : 'Internal server error' },
      { status: 500 }
    );
  }
}
```

```typescript
import { toast } from 'sonner';

interface TestSummary {
  totalPassed: number;
  totalFailed: number;
  passRate: number;
  meetsTarget: boolean;
  breakdown: {
    easy: { passed: number; total: number };
    medium: { passed: number; total: number };
    hard: { passed: number; total: number };
  };
  avgResponseTimeMs: number;
  avgSelfEvalScore: number;
}

interface TestResult {
  question: string;
  difficulty: string;
  passed: boolean;
  expectedAnswer: string;
  actualResponse: string;
  selfEvalScore: number | null;
  responseTimeMs: number;
  error?: string;
}

export default function GoldenSetTestPage() {
  const [isRunning, setIsRunning] = useState(false);
  const [summary, setSummary] = useState<TestSummary | null>(null);
  const [results, setResults] = useState<TestResult[]>([]);

  async function runTest() {
    setIsRunning(true);
    setSummary(null);
    setResults([]);

    try {
      const res = await fetch('/api/rag/test/golden-set', {
        method: 'POST',
      });

      if (!res.ok) {
        throw new Error(`Test failed: ${res.statusText}`);
      }

      const data = await res.json();

      if (data.success) {
        setSummary(data.data.summary);
        setResults(data.data.results);
        if (data.data.summary.meetsTarget) {
          toast.success(`Test passed! ${data.data.summary.passRate}% pass rate`);
        } else {
          toast.error(`Test failed: ${data.data.summary.passRate}% pass rate (target: 85%)`);
        }
      } else {
        toast.error(data.error || 'Test failed');
      }
    } catch (err) {
      toast.error(err instanceof Error ? err.message : 'Failed to run test');
    } finally {
      setIsRunning(false);
    }
  }

  return (
    <div className="container mx-auto py-8 space-y-6">
      <Card>
        <CardHeader>
          <CardTitle>RAG Golden-Set Regression Test</CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <p className="text-sm text-muted-foreground">
            Runs 20 Q&A pairs against the Sun Chip Bank document to validate RAG pipeline quality.
            Target: ≥85% pass rate (17/20 minimum).
          </p>

          <Button onClick={runTest} disabled={isRunning} size="lg">
            {isRunning ? (
              <>
                <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                Running test... (this takes ~2-3 minutes)
              </>
            ) : (
              <>
                <Play className="mr-2 h-4 w-4" />
                Run Golden-Set Test
              </>
            )}
          </Button>

          {summary && (
            <div className="mt-6 space-y-4">
              {/* Summary Card */}
              <Card className={summary.meetsTarget ? 'border-green-500' : 'border-red-500'}>
                <CardHeader>
                  <CardTitle className="flex items-center gap-2">
                    {summary.meetsTarget ? (
                      <CheckCircle2 className="h-5 w-5 text-green-600" />
                    ) : (
                      <XCircle className="h-5 w-5 text-red-600" />
                    )}
                    {summary.meetsTarget ? 'PASS' : 'FAIL'} — {summary.passRate}% pass rate
                  </CardTitle>
                </CardHeader>
                <CardContent className="grid grid-cols-2 md:grid-cols-4 gap-4">
                  <div>
                    <div className="text-2xl font-bold text-green-600">{summary.totalPassed}</div>
                    <div className="text-xs text-muted-foreground">Passed</div>
                  </div>
                  <div>
                    <div className="text-2xl font-bold text-red-600">{summary.totalFailed}</div>
                    <div className="text-xs text-muted-foreground">Failed</div>
                  </div>
                  <div>
                    <div className="text-2xl font-bold">{summary.avgResponseTimeMs}ms</div>
                    <div className="text-xs text-muted-foreground">Avg Response Time</div>
                  </div>
                  <div>
                    <div className="text-2xl font-bold">{(summary.avgSelfEvalScore * 100).toFixed(0)}%</div>
                    <div className="text-xs text-muted-foreground">Avg Self-Eval</div>
                  </div>
                </CardContent>
              </Card>

              {/* Breakdown by Difficulty */}
              <div className="grid grid-cols-3 gap-4">
                <Card>
                  <CardHeader className="pb-3">
                    <CardTitle className="text-sm">Easy</CardTitle>
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold">
                      {summary.breakdown.easy.passed}/{summary.breakdown.easy.total}
                    </div>
                  </CardContent>
                </Card>
                <Card>
                  <CardHeader className="pb-3">
                    <CardTitle className="text-sm">Medium</CardTitle>
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold">
                      {summary.breakdown.medium.passed}/{summary.breakdown.medium.total}

                  </CardContent>
                </Card>
                <Card>
                  <CardHeader className="pb-3">
                    <CardTitle className="text-sm">Hard</CardTitle>
                  </CardHeader>
                  <CardContent>

              {/* Detailed Results */}
              <Card>
                <CardHeader>
                  <CardTitle className="text-base">Detailed Results</CardTitle>
                </CardHeader>
                <CardContent>
                  <div className="space-y-2 max-h-[600px] overflow-y-auto">
                    {results.map((result, idx) => (
                      <div
                        key={idx}

                        }`}
                      >
                        <div className="flex items-start gap-2">
                          {result.passed ? (

                          </div>
                        </div>
                      </div>
                    ))}
                  </div>
                </CardContent>
              </Card>
            </div>
          )}
        </CardContent>
      </Card>
    </div>
  );
}
```

## Benefits of Vercel-Based Approach

1. **No cookie management** — uses server-side `requireAuth()`
2. **No dev server required** — runs on deployed Vercel instance
3. **Better error handling** — catches and reports errors per-query
4. **Visual UI** — see results in real-time with breakdown by difficulty
5. **Persistent** — can re-run anytime without CLI setup
6. **CI-ready** — can be called from GitHub Actions via HTTP POST

## Usage

1. Deploy to Vercel (or run locally with `npm run dev`)
2. Navigate to `/rag/test`
3. Click "Run Golden-Set Test"
4. Wait ~2-3 minutes for all 20 queries to execute
5. View pass/fail summary and detailed results
